import numpy as np
import random

class SovereignNode:
    def __init__(self, node_id, beta=2.5, k_factor=0.1, decay=0.95):
        self.id = node_id
        self.state = 1.0
        self.persistence = 1.0
        self.beta = beta
        self.k_factor = k_factor
        self.decay = decay
        self.neighbors = []

    def add_neighbor(self, neighbor):
        if neighbor not in self.neighbors:
            self.neighbors.append(neighbor)

    def update(self):
        if not self.neighbors: return
        field = sum(n.state for n in self.neighbors)

            self.state *= -1
            self.persistence *= self.decay
        else:
            self.persistence += self.k_factor * (1.0 - self.persistence)
            self.persistence = min(1.0, self.persistence)
